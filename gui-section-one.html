java swing 的 hello world

package com.gui.start;

/**
 * @author Hester
 * 初始的frame运用，产生的frame只有标题
 */

import javax.swing.*;

public class HelloSwing {
    public static void main(String [] args) {
        JFrame frame = new JFrame("Hello Swing"); //新建一个frame对象并设置标题
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); //激活关闭按钮
        frame.setSize(300, 160); //设置Frame的大小
        frame.setVisible(true); //设置为可见
        for(int i = 0; i < (int)(Math.random() * 20); i++) {
            System.out.println(i);
            JLabel label = new JLabel("label No." + i);
            frame.add(label);
        }
    }
}

修改后的Swing程序

package com.gui.start;

import javax.swing.*;
import java.util.concurrent.*;

//正规的GUI编程方式，继承Frame类并且main线程只负责分发任务
//:com/gui/start/SubmitSwingPrograme.java
public class SubmitSwingPrograme extends JFrame {
    JLabel label;
    public SubmitSwingPrograme() {
        super("Hello Swing!");
        label = new JLabel("A Label");
        this.add(label);
    }

    public static void main(String[] args) {
        // TODO Auto-generated method stub
        final SubmitSwingPrograme ssp = new SubmitSwingPrograme();
        SwingConsole.run(ssp, 1080, 720);
        try {
            TimeUnit.SECONDS.sleep(1);
        } catch (InterruptedException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                ssp.label.setText("This is different!");
            }
        });
    }

}

//:com/gui/start/SwingConsole.java
package com.gui.start;

import javax.swing.*;

public class SwingConsole {
    public static void run(final JFrame frame, final int width, final int height) {
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                frame.setTitle(frame.getClass().getSimpleName());
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
                frame.setSize(width, height);
                frame.setVisible(true);
            }
        });
    }
}

一般GUI编程主类继承Frame，同时在构造方法中设置基本的参数如大小、标题等，同时要设置成可见 main线程中创建实现Runnable的匿名类来改变Frame中的元素如Label、Button等。 final 关键字用于变量的理解：将变量设置为常量，即只读，或者调用其方法，无法对其赋值。
